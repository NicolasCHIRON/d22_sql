CREATE TABLE `Pins` (
  `ID` INTEGER,
  `url` TEXT,
  PRIMARY KEY (`ID`)
);

CREATE TABLE `Users` (
  `ID` INTEGER,
  `name` TEXT,
  PRIMARY KEY (`ID`)
);

CREATE TABLE `Comments` (
  `ID` INTEGER,
  `pinID` INTEGER,
  `userID` INTEGER,
  `text` TEXT,
  PRIMARY KEY (`ID`),
  FOREIGN KEY (`pinID`) REFERENCES `Pins`(`ID`),
  FOREIGN KEY (`userID`) REFERENCES `Users`(`ID`)
);
